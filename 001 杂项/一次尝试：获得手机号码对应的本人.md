#### 通过正则表达式，取得一行中的前缀7位部分
~~~
import re

# Open the file for reading
with open("beijing.txt", "r", encoding="utf-8") as file:
    # Iterate through each line in the file
    for line in file:
        # Use re.search to find the 7-digit number in the line
        match = re.search(r"'(\d{7})'", line)
        
        # If a match is found, print the matched group
        if match:
            print(match.group(1))

~~~

#### 基于前缀和后缀，发送api请求
~~~
import requests
import time
import json

# 读取生成的手机号
with open("1510100.txt", "r") as f:
    numbers = [num.strip() for num in f.readlines()]

results = []

for number in numbers:
    for middle in range(10):  # 中间位遍历0-9
        full_number = f"{number}{middle}885"
	    url = f"https://zy.xywlapi.cc/wbphone?phone={number}"
	    try:
	        resp = requests.get(url)
	        data = resp.json()
	        data["queried_number"] = number  # 记录查询的原始号码
	        results.append(data)
	        print(f"已查询: {number}")
	        time.sleep(0.5)  # 礼貌性延时
	    except Exception as e:
	        print(f"查询失败 {number}: {str(e)}")

# 保存为JSON文件
with open("result.json", "w", encoding="utf-8") as f:
    json.dump(results, f, ensure_ascii=False, indent=2)
